 21def parse_correlation_self_energy_params(file_string: str) -> dict:
 22    """Parse correlation self-energy parameters.
 23
 24    Match the key on one line and return the value from the second, according to some
 25    specified extraction behaviour given by `keys_extractions`.
 26
 27    Also extract the analytic continuation and singularity schemes, plus their references.
 28
 29    :param str file_string: Input string
 30    :return dict data: Matched data
 31    """
 32    keys_extractions = {'Solution of the QP equation': lambda x: int(x.split()[0]),
 33                        'Energy alignment': lambda x: int(x.split()[0]),
 34                        'Analytic continuation method': lambda x: x.strip(),
 35                        'Scheme to treat singularities': lambda x: x.strip()
 36                        }
 37
 38    data = match_current_extract_from_line_n(file_string, keys_extractions)
 39
 40    # Also extract citations
 41    def ref_after_citation(x: str) -> str:
 42        """
 43        Expected format is
 44        Citation: Paper reference
 45        """
 46        try:
 47            i = x.index(":")
 48            return x[i + 1:].strip()
 49        except ValueError:
 50            return 'Not found'
 51
 52    # Extraction of citations after
 53    for key in ['Analytic continuation method', 'Scheme to treat singularities']:
 54        citation_extraction = {key: ref_after_citation}
 55        matched_dictionary = match_current_extract_from_line_n(file_string, citation_extraction, n_line=2)
 56        data[key + ' citation'] = matched_dictionary[key]
 57
 58    return data

 61def parse_mixed_product_params(file_string: str) -> dict:
 62    """Parse mixed product basis parameters.
 63
 64    :param str file_string: Input string
 65    :return dict data: Matched data
 66    """
 67    search_keys = [
 68        'Angular momentum cutoff:', 'Linear dependence tolerance factor:',
 69        'Plane wave cutoff \\(in units of Gkmax\\):'
 70    ]
 71
 72    data = parse_values_regex(file_string, search_keys)
 73
 74    # Rename keys to give better context
 75    modified_data = {
 76        'MT Angular momentum cutoff':
 77        data['Angular momentum cutoff'],
 78        'MT Linear dependence tolerance factor':
 79        data['Linear dependence tolerance factor'],
 80        'Plane wave cutoff (in units of Gkmax)':
 81        data['Plane wave cutoff (in units of Gkmax)']
 82    }
 83
 84    return modified_data

 87def parse_bare_coulomb_potential_params(file_string: str) -> dict:
 88    """Extract bare Coulomb parameter data of the form:
 89
 90    Bare Coulomb potential parameters:
 91        Plane wave cutoff (in units of Gkmax*input%gw%MixBasis%gmb): 2.00000000000000
 92        Error tolerance for structure constants:   1.000000000000000E-016
 93        Tolerance factor to reduce the MB size based on
 94        the eigenvectors of the bare Coulomb potential:   0.100000000000000
 95
 96    and return a dictionary of the form:
 97
 98    {'Plane wave cutoff (in units of Gkmax*gmb)': 2.0,
 99     'Error tolerance for structure constants': 1e-16,
100      'MB tolerance factor': 0.1
101    }
102
103    :param str file_string: Input string
104    :return dict data: Matched data
105    """
106
107    pw_cutoff_matching_str = 'Plane wave cutoff \\(in units of Gkmax\\*input%gw%MixBasis%gmb\\):'
108    pw_cutoff = list(parse_value_regex(file_string, pw_cutoff_matching_str).values())
109    assert len(pw_cutoff) == 1, "Matched plane wave cutoff is ambiguous - more than one match"
110
111    # Defined with less-verbose key
112    data = {'Plane wave cutoff (in units of Gkmax*gmb)': float(pw_cutoff[0])}
113
114    data2 = parse_value_regex(file_string, 'Error tolerance for structure constants:')
115
116    data3 = parse_value_regex(file_string, 'the eigenvectors of the bare Coulomb potential:')
117
118    # Defined with less-verbose key (see docs above for full key, over 2 lines)
119    modified_data3 = {'MB tolerance factor': data3.pop('the eigenvectors of the bare Coulomb potential')}
120
121    return {**data, **data2, **modified_data3}

155def parse_frequency_grid(file_string: str, n_points: int) -> np.ndarray:
156    """Parse the frequency grid and weights used with GW
157
158    :param str file_string: Input string
159    :param int n_points: Number of frequency grid points
160    :return np.ndarray grid_and_weights: Frequency grid and weights in grid[0, :] and grid[1, :],
161    respectively
162    """
163    index = file_string.find('frequency list: < #    freqs    weight >')
164    frequency_lines = file_string[index:].split('\n')
165
166    grid_and_weights = np.empty(shape=(2, n_points))
167    for i in range(0, n_points):
168        index, frequency, weight = frequency_lines[i + 1].split()
169        grid_and_weights[0:2, i] = np.array([float(frequency), float(weight)])
170
171    return grid_and_weights

174def parse_ks_eigenstates(file_string: str) -> dict:
175    """Parse information on the KS eigenstates used.
176    Note, final keys will have the trailing whitespace stripped.
177
178    :param str file_string: Input string
179    :return dict data: Matched data
180    """
181
182    # Trailing whitespace in some instances required for match
183    ks_eigenstates_keys = ['Maximum number of LAPW states:',
184                           'Minimal number of LAPW states:',
185                           '- total KS',
186                           '- occupied',
187                           '- unoccupied ',
188                           '- dielectric function',
189                           '- self energy',
190                           'Energy of the highest unoccupied state: ',
191                           'Number of valence electrons:',
192                           'Number of valence electrons treated in GW: '
193                           ]
194
195    data = parse_values_regex(file_string, ks_eigenstates_keys)
196
197    # Prepend keys that lack context, such as '- total KS'
198    modified_data = {}
199    prepend_str = "Number of states used in GW"
200
201    for key, value in data.items():
202        if key[0] == '-':
203            new_key = prepend_str + " " + key.rstrip().rstrip(':')
204        else:
205            new_key = key.rstrip().rstrip(':')
206        modified_data[new_key] = value
207
208    return modified_data

211def parse_n_q_point_cycles(file_string: str) -> int:
212    """Get the maximum number of q iterations performed.
213

223def extract_kpoint(file_string: str) -> dict:
224    """Parse the substring of the form:
225
226     at k =    0.000   0.500   0.500 ik =     3
227
228    returning a dictionary of the form:
229
230     {'VBM': {'k_point': [0.0, 0.5, 0.5], 'ik': 3},
231      'CBm': {'k_point': [0.0, 0.5, 0.5], 'ik': 3}
232      }
233
234    :param str file_string: Input string
235    :return dict k_data: Matched data, of the form documented
236    """
237
238    def parse_k_match(file_string: str, key: str) -> dict:
239        data = {}
240
241        try:
242            match = re.search(key + '(.+?)\n', file_string)
243            k_point_and_index = match.group(1).split()
244            data = {'k_point': [float(k) for k in k_point_and_index[:3]],
245                    'ik': int(k_point_and_index[-1])}
246
247        except AttributeError:
248            raise AttributeError("extract_kpoint. Did not find the key", match)
249
250        return data
251
252    k_data = parse_k_match(file_string, 'at k      =    ')
253
254    return {'VBM': k_data, 'CBm': k_data}

257def extract_kpoints(file_string: str) -> dict:
258    """Parse the substring of the form:
259
260     at k(VBM) =    0.000   0.500   0.500 ik =     3
261        k(CBm) =    0.000   0.000   0.000 ik =     1
262
263    returning a dictionary of the form:
264
265     {'VBM': {'k_point': [0.0, 0.5, 0.5], 'ik': 3},
266      'CBm': {'k_point': [0.0, 0.0, 0.0], 'ik': 1}
267      }
268
269    :param str file_string: Input string
270    :return dict k_data: Matched data, of the form documented
271    """
272
273    def parse_k_match(file_string: str, key: str) -> dict:
274        data = {}
275
276        match_key_to_parser_key = {
277            'at k\\(VBM\\) = ': 'VBM',
278            'k\\(CBm\\) = ': 'CBm'
279        }
280
281        try:
282            match = re.search(key + '(.+?)\n', file_string)
283            parser_key = match_key_to_parser_key[key].replace('\\', "")
284            k_point_and_index = match.group(1).split()
285            data[parser_key] = {'k_point': [float(k) for k in k_point_and_index[:3]],
286                                'ik': int(k_point_and_index[-1])}
287
288        except AttributeError:
289            print("extract_kpoints. Did not find the key", match)
290
291        return data
292
293    k_data = parse_k_match(file_string, 'at k\\(VBM\\) = ')
294    k_data2 = parse_k_match(file_string, 'k\\(CBm\\) = ')
295
296    return {** k_data, **k_data2}

299def parse_band_structure_info(file_string: str, bs_type: str) -> dict:
300    """Parse KS or GW band structure information.
301
302    This routine assumes that the KS band structure info will ALWAYS appear
303    before the GW band structure info.
304
305    Two situations can occur.
306
307    * 1. Indirect bandgap:
308
309     Indirect BandGap (eV):                    3.3206
310     at k(VBM) =    0.000   0.500   0.500 ik =     3
311        k(CBm) =    0.000   0.000   0.000 ik =     1
312     Direct Bandgap at k(VBM) (eV):            3.7482
313     Direct Bandgap at k(CBm) (eV):            3.8653
314
315    * 2. Direct bandgap:
316
317      Direct BandGap (eV):                      2.3903
318      at k      =    0.000   0.000   0.000 ik =     1
319
320    :param str bs_type: Band structure type to parse. Either 'ks' or 'gw'
321    :param str file_string: Input string
322    :return dict k_data: Matched data
323    """
324    if bs_type == 'ks':
325        # Parse first instance of each key, exploiting that Kohn-Sham band structure
326        # comes before G0W0 band structure. This ASSUMES fixed structure to the file
327        pass
328
329    elif bs_type == 'gw':
330        # Find G0W0 band structure in the file, then start parsing from there
331        gw_header = ' G0W0 band structure '
332        index = file_string.find('G0W0 band structure ')
333        file_string = file_string[index:]
334
335    else:
336        sys.exit("bs_type must be 'ks' or 'gw'")
337
338    # Indirect BandGap may not be present
339    band_structure_keys = ['Fermi energy:',
340                           'Energy range:',
341                           'Band index of VBM:',
342                           'Band index of CBm:']
343
344    data = parse_values_regex(file_string, band_structure_keys)
345
346    # Only present if there's an indirect gap
347    indirect_gap = parse_value_regex(file_string,
348                                     'Indirect BandGap \\(eV\\):',
349                                     silent_key_error=True)
350
351    if indirect_gap:
352        direct_gap_keys = [
353            'Direct Bandgap at k\\(VBM\\) \\(eV\\):',
354            'Direct Bandgap at k\\(CBm\\) \\(eV\\):'
355        ]
356        data.update(indirect_gap)
357        data.update(parse_values_regex(file_string, direct_gap_keys))
358        k_point_data = extract_kpoints(file_string)
359
360    else:
361        data.update(parse_value_regex(file_string, 'Direct BandGap \\(eV\\):'))
362        k_point_data = extract_kpoint(file_string)
363
364    return {**data, **k_point_data}

368def parse_gw_info(file_string: str) -> dict:
369    """Parse data from GW_INFO.OUT.
370
371    Timings are not parsed because it's more convenient for regression-testing.
372
373    :param str file_string: Parsed file string.
374    :return: dict data: dictionary of parsed data.
375    """
376    data = {}
377    data['correlation_self_energy_parameters'] = parse_correlation_self_energy_params(file_string)
378    data['mixed_product_basis_parameters'] = parse_mixed_product_params(file_string)
379    data['bare_coulomb_potential_parameters'] = parse_bare_coulomb_potential_params(file_string)
380    data['screened_coulomb_potential'] =  match_current_return_line_n(file_string, 'Screened Coulomb potential:').strip()
381    data['core_electrons_treatment'] = match_current_return_line_n(file_string, 'Core electrons treatment:').strip()
382    data['qp_interval'] = parse_value_regex(file_string, 'Interval of quasiparticle states \\(ibgw, nbgw\\):')\
383        ['Interval of quasiparticle states (ibgw, nbgw)']
384    data['n_empty'] = parse_value_regex(file_string, 'Number of empty states \\(GW\\):')['Number of empty states (GW)']
385    data['q_grid'] = parse_value_regex(file_string, 'k/q-points grid:')['k/q-points grid']
386    data['mixed_product_wf_info'] = parse_mixed_product_wf_info(file_string)
387    data['frequency_grid'] = parse_frequency_grid_info(file_string)
388    n_freq_points = data['frequency_grid']['Number of frequencies: < nomeg >']
389    data['frequency_grid']['frequencies_weights'] = parse_frequency_grid(file_string, n_freq_points)
390    data['ks_eigenstates_summary'] = parse_ks_eigenstates(file_string)
391    data['ks_band_structure_summary'] = parse_band_structure_info(file_string, 'ks')
392    data['n_q_cycles'] = parse_n_q_point_cycles(file_string)
393    data['g0w0_band_structure_summary'] = parse_band_structure_info(file_string, 'gw')
394    return data

397def extract_gw_timings_as_list(file_string: str) -> Union[List[str], None]:
398    """Extract GW timing string block as a list.
399
400    Utilises the fact that timings are returned at the end of GW_INFO.OUT
401
402    :param str file_string: Parsed file string
403    :param list timings: GW timings, with each element storing a
404    line of timings as a string.
405    """
406    file_list = file_string.split('\n')
407    for i, line in enumerate(reversed(file_list)):
408        if 'GW timing info (seconds)' in line:
409            index = i - 2
410            return file_list[-index:]
411
412    return None

415def parse_gw_timings(file_string: str) -> dict:
416    """Parse timings returned by the GW method.
417
418    Assumptions:
419     * Initial timing value is 'Initialization'
420     * Final timing value is 'Total'
421
422    Returns a dictionary with each time subheading storing
423    a dictionary of timing breakdowns.
424
425    If the subheading has a time associated with it, that is stored
426    in the breakdowns with the same key. If it has no associated timing,
427    the value is None.
428
429    For example:
430
431    'Subroutines': {'Subroutines': None,
432                    'calcpmat': 5.12,
433                    'calcbarcmb': 5.65,
434                    'BZ integration weights': 18.35
435                    }
436
437    :param str file_string: Parsed file string
438    :return: dict data: dictionary of parsed timings
439    """
440    timings = extract_gw_timings_as_list(file_string)
441
442    # Parse and store nested timings
443    data = {}
444    component_time = {}
445    root_key = 'Initialization'
446    initial_key = root_key
447
448    for line in timings:
449
450        key = line.strip().split(':')[0].rstrip()
451        if len(key) == 0: continue
452        is_root_key = key[0] != '-'
453
454        if is_root_key and (key != initial_key):
455            data[root_key] = copy.deepcopy(component_time)
456            component_time.clear()
457            root_key = key
458
459        time_str = line.strip().split(':')[-1]
460        component_time[key.strip('-').strip()] = float(
461            time_str) if can_be_float(time_str) else None
462
463    # Store last value, Total
464    data[root_key] = {root_key: float(time_str)}
465
466    # Remove superfluous key from table formatting
467    arb_str = '_________________________________________________________'
468    del data[arb_str]
469
470    return data
